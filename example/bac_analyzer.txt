class AnalysisLogger:
    """Logger for capturing all Claude interactions"""
    
    def __init__(self, analysis_id: str, base_dir: Path):
        self.analysis_id = analysis_id
        self.log_dir = base_dir / analysis_id / "logs"
        self.log_dir.mkdir(exist_ok=True, parents=True)
        
        # Setup file handlers
        self.query_log = self.log_dir / "queries.log"
        self.response_log = self.log_dir / "responses.log"
        self.error_log = self.log_dir / "errors.log"
        self.debug_log = self.log_dir / "debug.log"
        
        # Python logging setup
        self.logger = logging.getLogger(f"analysis_{analysis_id}")
        self.logger.setLevel(logging.DEBUG)
        
        # Remove existing handlers to avoid duplicates
        self.logger.handlers = []
        
        fh = logging.FileHandler(self.debug_log)
        fh.setLevel(logging.DEBUG)
        formatter = logging.Formatter('%(asctime)s - %(levelname)s - %(message)s')
        fh.setFormatter(formatter)
        self.logger.addHandler(fh)
        
        # Also log to console
        ch = logging.StreamHandler()
        ch.setLevel(logging.INFO)
        ch.setFormatter(formatter)
        self.logger.addHandler(ch)
    
    def log_query(self, prompt: str):
        """Log Claude query"""
        timestamp = datetime.now().isoformat()
        with open(self.query_log, 'a') as f:
            f.write(f"\n{'='*80}\n")
            f.write(f"QUERY AT {timestamp}\n")
            f.write(f"{'='*80}\n")
            f.write(prompt)
            f.write(f"\n{'='*80}\n\n")
        self.logger.info(f"Sent query ({len(prompt)} chars)")
    
    def log_response(self, response: str):
        """Log Claude response"""
        timestamp = datetime.now().isoformat()
        with open(self.response_log, 'a') as f:
            f.write(f"\n{'='*80}\n")
            f.write(f"RESPONSE AT {timestamp}\n")
            f.write(f"{'='*80}\n")
            f.write(response)
            f.write(f"\n{'='*80}\n\n")
        self.logger.info(f"Received response ({len(response)} chars)")
    
    def log_error(self, error: Exception, context: str = ""):
        """Log error with full traceback"""
        timestamp = datetime.now().isoformat()
        tb = traceback.format_exc()
        
        with open(self.error_log, 'a') as f:
            f.write(f"\n{'='*80}\n")
            f.write(f"ERROR AT {timestamp}\n")
            f.write(f"Context: {context}\n")
            f.write(f"{'='*80}\n")
            f.write(f"Error: {str(error)}\n")
            f.write(f"Traceback:\n{tb}\n")
            f.write(f"{'='*80}\n\n")
        self.logger.error(f"{context}: {str(error)}")
    
    def log_message(self, message: str, level: str = "info"):
        """Log a general message"""
        if level == "debug":
            self.logger.debug(message)
        elif level == "warning":
            self.logger.warning(message)
        elif level == "error":
            self.logger.error(message)
        else:
            self.logger.info(message)


class ClaudeBACAnalyzer:
    """
    Interactive BAC/IDOR analyzer using ClaudeSDKClient.
    
    Features:
    - Detects authorization mechanisms
    - Requests additional user tokens when needed
    - Generates and executes BAC/IDOR tests
    - Analyzes responses for access control violations
    """
    
    def __init__(self, work_dir: str = None):
        """
        Initialize Claude BAC Analyzer.
        
        Args:
            work_dir: Working directory for Claude sessions
        """
        self.work_dir = Path(work_dir or os.getenv("CLAUDE_WORK_DIR", "./analysis_workdir"))
        self.work_dir.mkdir(exist_ok=True, parents=True)
        self.prompts_dir = Path(__file__).parent.parent.parent / "prompts"
        self.logger = None  # Will be set per analysis
    
    async def analyze(self, analysis: Analysis) -> Analysis:
        """
        Perform BAC/IDOR analysis on a request.
        
        Args:
            analysis: Analysis object with initial request
            
        Returns:
            Updated analysis object with results or data requests
        """
        # Create logger for this analysis
        self.logger = AnalysisLogger(analysis.id, self.work_dir)
        
        if not ClaudeSDKClient:
            analysis.status = AnalysisStatus.FAILED
            analysis.error = "ClaudeSDKClient not available"
            self.logger.log_error(Exception("ClaudeSDKClient not available"), "Initialization")
            return analysis
        
        # Create session working directory
        session_dir = self.work_dir / analysis.id
        session_dir.mkdir(exist_ok=True)
        
        self.logger.log_message("Starting BAC/IDOR analysis")
        self.logger.log_message(f"Target: {analysis.target_url}", "debug")
        self.logger.log_message(f"Method: {analysis.request.method}", "debug")
        
        # Save HTTP client tool for Claude to use
        self._setup_http_tool(session_dir)
        self.logger.log_message("HTTP tool setup complete", "debug")
        
        try:
            # Add Claude CLI to PATH if needed
            claude_cli_dir = os.path.expanduser(
                os.getenv("CLAUDE_CLI_DIR", "~/.claude/local")
            )
            if claude_cli_dir and os.path.exists(claude_cli_dir):
                os.environ["PATH"] = f"{claude_cli_dir}:{os.environ.get('PATH', '')}"
                self.logger.log_message(f"Added to PATH: {claude_cli_dir}", "debug")
            
            # Configure Claude options with MCP server for HTTP requests
            mcp_server_path = str(Path(__file__).parent.parent.parent / "integrations" / "mcp_server.py")
            
            options = ClaudeCodeOptions(
                cwd=str(session_dir),
                allowed_tools=[
                    "Bash", "Glob", "Grep", "LS", "Read", "Write",
                    "Edit", "MultiEdit", "exit_plan_mode",
                    # MCP tools for HTTP requests
                    "mcp__paintester-http__http_request",
                    "mcp__paintester-http__send_bac_test",
                    "mcp__paintester-http__send_idor_test",
                    "mcp__paintester-http__compare_responses"
                ],
                permission_mode="acceptEdits",
                mcp_servers={
                    "paintester-http": {
                        "command": "python",
                        "args": [mcp_server_path],
                        "env": {}
                    }
                }
            )
            
            # Create Claude session
            self.logger.log_message("Creating Claude session")
            async with ClaudeSDKClient(options=options) as client:
                analysis.claude_session_id = str(id(client))
                self.logger.log_message(f"Claude session created: {analysis.claude_session_id}", "debug")
                
                # Initial analysis phase
                prompt = self._build_initial_prompt(analysis)
                self.logger.log_query(prompt)
                
                self.logger.log_message("Sending query to Claude")
                await client.query(prompt)
                
                # Process Claude's response
                self.logger.log_message("Collecting response from Claude")
                response_text = await self._collect_response(client)
                self.logger.log_response(response_text)
                
                self.logger.log_message("Parsing Claude response")
                result = self._parse_claude_response(response_text)
                
                # Handle different outcomes
                if result.get("status") == "need_more_data":
                    self.logger.log_message("Claude needs more data")
                    analysis = self._handle_data_request(analysis, result)
                    self._save_analysis_state(analysis, session_dir)
                    return analysis
                
                elif result.get("status") == "ready_to_test":
                    self.logger.log_message("Claude ready to test")
                    # Extract auth profile
                    if "auth_profile" in result:
                        analysis.auth_profile = AuthProfile(**result["auth_profile"])
                        self.logger.log_message(f"Auth profile extracted: {analysis.auth_profile.auth_type}", "debug")
                    
                    # Proceed to testing phase
                    analysis = await self._execute_testing_phase(client, analysis, result)
                    self._save_analysis_state(analysis, session_dir)
                    return analysis
                
                else:
                    # Unexpected response
                    self.logger.log_message(f"Unexpected status: {result.get('status', 'unknown')}", "warning")
                    analysis.ai_analysis = response_text
                    analysis.status = AnalysisStatus.COMPLETED
                    self._save_analysis_state(analysis, session_dir)
                    return analysis
        
        except Exception as e:
            self.logger.log_error(e, "Analysis failed during Claude interaction")
            analysis.status = AnalysisStatus.FAILED
            analysis.error = f"{type(e).__name__}: {str(e)}"
            self._save_analysis_state(analysis, session_dir)
            return analysis

    async def continue_with_context(self, analysis: Analysis) -> Analysis:
        """
        Continue analysis with user-provided text context (incremental mode).
        Uses three-phase deterministic testing, comparing new results against previous findings.
        
        Phase 1: Claude generates test cases considering user context
        Phase 2: Python executes tests deterministically via httpx
        Phase 3: Claude analyzes results and compares with previous findings
        
        Args:
            analysis: Analysis with user_context set
            
        Returns:
            Updated analysis object with refined findings
        """
        self.logger = AnalysisLogger(analysis.id, self.work_dir)
        
        if not ClaudeSDKClient:
            analysis.status = AnalysisStatus.FAILED
            analysis.error = "ClaudeSDKClient not available"
            self.logger.log_error(Exception("ClaudeSDKClient not available"), "Continue with Context")
            return analysis
        
        session_dir = self.work_dir / analysis.id
        session_dir.mkdir(exist_ok=True)
        self.logger.log_message("Continuing analysis with user-provided text context (3-phase deterministic)")
        self.logger.log_message(f"User context: {analysis.user_context[:200]}..." if analysis.user_context and len(analysis.user_context) > 200 else f"User context: {analysis.user_context}", "debug")
        self.logger.log_message(f"Previous findings count: {len(analysis.findings)}")
        
        # Store previous findings for comparison
        previous_findings = list(analysis.findings)
        
        try:
            # === PHASE 1: Generate Test Plan with User Context ===
            self.logger.log_message("=== PHASE 1: Generating test plan with user context ===")
            test_plan = await self._generate_test_plan_with_context(analysis, session_dir)
            
            if not test_plan or not test_plan.get("test_cases"):
                self.logger.log_message("No test cases generated, keeping previous findings")
                analysis.status = AnalysisStatus.COMPLETED
                analysis.completed_at = datetime.utcnow()
                self._save_analysis_state(analysis, session_dir)
                return analysis
            
            self.logger.log_message(f"Generated {len(test_plan.get('test_cases', []))} test cases")
            
            # Save test plan
            test_plan_path = session_dir / "test_cases.json"
            with open(test_plan_path, 'w') as f:
                json.dump(test_plan, f, indent=2)
            
            # === PHASE 2: Execute Tests Deterministically ===
            self.logger.log_message("=== PHASE 2: Executing tests (no AI involvement) ===")
            
            from engine.core.test_executor import TestExecutor
            executor = TestExecutor(analysis.id, self.work_dir)
            test_results = executor.execute_all_tests()
            
            if not test_results or not test_results.get("test_results"):
                self.logger.log_message("No test results generated, keeping previous findings")
                analysis.status = AnalysisStatus.COMPLETED
                analysis.completed_at = datetime.utcnow()
                self._save_analysis_state(analysis, session_dir)
                return analysis
            
            self.logger.log_message(f"Executed {len(test_results.get('test_results', []))} tests")
            
            # === PHASE 3: Analyze Results with Previous Findings Context ===
            self.logger.log_message("=== PHASE 3: Analyzing test results (incremental mode) ===")
            analysis = await self._analyze_test_results_incremental(
                analysis, test_results, previous_findings, session_dir
            )
            
            # Attach raw request/response to verified findings
            for finding in analysis.findings:
                if finding.source == "verified" and finding.verified_by_test_id:
                    test_result = next(
                        (t for t in test_results.get("test_results", [])
                         if t.get("test_id") == finding.verified_by_test_id),
                        None
                    )
                    if test_result:
                        finding.poc_request_raw = test_result.get("request_raw")
                        finding.poc_response_raw = test_result.get("response_raw")
                        self.logger.log_message(
                            f"Attached Burp-friendly PoC to finding: {finding.title[:50]}",
                            "debug"
                        )
            
            analysis.status = AnalysisStatus.COMPLETED
            analysis.completed_at = datetime.utcnow()
            self._save_analysis_state(analysis, session_dir)
            
            verified_count = sum(1 for f in analysis.findings if f.source == 'verified')
            fp_count = sum(1 for f in analysis.findings if f.false_positive)
            self.logger.log_message(
                f"Incremental analysis complete: {len(analysis.findings)} findings "
                f"({verified_count} verified, {fp_count} marked as FP)"
            )
            
            return analysis
        
        except Exception as e:
            self.logger.log_error(e, "Context continuation failed")
            analysis.status = AnalysisStatus.FAILED
            analysis.error = f"Context continuation failed: {str(e)}"
            self._save_analysis_state(analysis, session_dir)
            return analysis
    
    async def _generate_test_plan_with_context(
        self,
        analysis: Analysis,
        session_dir: Path
    ) -> Dict[str, Any]:
        """
        Generate test cases with user-provided context.
        Similar to _generate_test_plan but includes user context in the prompt.
        
        Args:
            analysis: Analysis object with user_context and captured traffic
            session_dir: Session directory
            
        Returns:
            Test plan dict with test_cases array
        """
        if not ClaudeSDKClient:
            raise Exception("ClaudeSDKClient not available")
        
        # Add Claude CLI to PATH if needed
        claude_cli_dir = os.path.expanduser(
            os.getenv("CLAUDE_CLI_DIR", "~/.claude/local")
        )
        if claude_cli_dir and os.path.exists(claude_cli_dir):
            os.environ["PATH"] = f"{claude_cli_dir}:{os.environ.get('PATH', '')}"
        
        # Configure Claude options - NO MCP tools for test generation
        options = ClaudeCodeOptions(
            cwd=str(session_dir),
            allowed_tools=["Bash", "Glob", "Grep", "LS", "Read"],
            permission_mode="acceptEdits"
        )
            
        # Build prompt with user context
        prompt_template = self._load_prompt("bac_generate_tests")
        
        # Format captured traffic
        if analysis.request_response_pairs:
            request_info, response_info = self._format_request_response_pairs(analysis)
        else:
            request_info = self._format_single_request(analysis.request)
            response_info = self._format_single_response(analysis.response) if analysis.response else ""
        
        prompt = prompt_template.replace("{captured_traffic}", request_info + "\n" + response_info)
        prompt = prompt.replace("{subdomain}", analysis.subdomain or "unknown")
        prompt = prompt.replace("{analysis_id}", analysis.id)
        prompt = prompt.replace("{user_context}", analysis.user_context or "(No additional context provided)")
        
        self.logger.log_query(prompt)
        
        async with ClaudeSDKClient(options=options) as client:
            await client.query(prompt)
            response_text = await self._collect_response(client)
            self.logger.log_response(response_text)
        
        # Parse test plan
        test_plan = self._parse_claude_response(response_text)
        
        # Fallback: check if Claude wrote to a file instead
        if not test_plan.get("test_cases"):
            test_plan_file = session_dir / "test_plan.json"
            if test_plan_file.exists():
                self.logger.log_message("Found test_plan.json file, loading from disk", "debug")
                try:
                    with open(test_plan_file) as f:
                        test_plan = json.load(f)
                except (json.JSONDecodeError, IOError) as e:
                    self.logger.log_message(f"Failed to load test_plan.json: {e}", "warning")
        
        return test_plan
    
    async def _analyze_test_results_incremental(
        self,
        analysis: Analysis,
        test_results: Dict[str, Any],
        previous_findings: list,
        session_dir: Path
    ) -> Analysis:
        """
        Phase 3 (Incremental): Analyze test results and compare with previous findings.
        
        Handles three actions from Claude:
        - mark_fp: Mark a previous finding as false positive
        - update: Update an existing finding with new evidence
        - create: Create a new finding
        
        Args:
            analysis: Analysis object
            test_results: Results from TestExecutor
            previous_findings: List of findings from previous analysis
            session_dir: Session directory
            
        Returns:
            Updated analysis with refined findings
        """
        if not ClaudeSDKClient:
            raise Exception("ClaudeSDKClient not available")
        
        # Add Claude CLI to PATH if needed
        claude_cli_dir = os.path.expanduser(
            os.getenv("CLAUDE_CLI_DIR", "~/.claude/local")
        )
        if claude_cli_dir and os.path.exists(claude_cli_dir):
            os.environ["PATH"] = f"{claude_cli_dir}:{os.environ.get('PATH', '')}"
        
        # Configure Claude options - NO MCP tools for result analysis
        options = ClaudeCodeOptions(
            cwd=str(session_dir),
            allowed_tools=["Bash", "Glob", "Grep", "LS", "Read", "Write", "Edit"],
            permission_mode="acceptEdits"
        )
        
        # Build prompt
        prompt_template = self._load_prompt("bac_analyze_results_incremental")
        
        # Format test results
        test_results_json = json.dumps(test_results, indent=2)
        
        # Format previous findings for the prompt
        previous_findings_json = json.dumps(
            [
                {
                    "id": f.id,
                    "title": f.title,
                    "vulnerability_type": f.vulnerability_type,
                    "severity": f.severity,
                    "description": f.description,
                    "affected_url": f.affected_url,
                    "source": f.source,
                    "verified_by_test_id": f.verified_by_test_id
                }
                for f in previous_findings
            ],
            indent=2
        )
        
        prompt = prompt_template.replace("{test_results}", test_results_json)
        prompt = prompt.replace("{analysis_id}", analysis.id)
        prompt = prompt.replace("{previous_findings}", previous_findings_json)
        prompt = prompt.replace("{user_context}", analysis.user_context or "(No additional context provided)")
        
        self.logger.log_query(prompt)
        
        async with ClaudeSDKClient(options=options) as client:
            await client.query(prompt)
            response_text = await self._collect_response(client)
            self.logger.log_response(response_text)
        
        # Parse results
        result = self._parse_claude_response(response_text)
        analysis.ai_analysis = response_text
        
        # Create a map of previous findings by ID for quick lookup
        findings_map = {f.id: f for f in previous_findings}
        
        # Process findings with actions
        if "findings" in result and result["findings"]:
            for finding_data in result["findings"]:
                action = finding_data.get("action", "create")
                
                if action == "mark_fp":
                    # Mark existing finding as false positive
                    prev_id = finding_data.get("previous_finding_id")
                    if prev_id and prev_id in findings_map:
                        fp_finding = findings_map[prev_id]
                        fp_finding.false_positive = True
                        fp_finding.false_positive_reason = finding_data.get("fp_reason", "Marked as FP during incremental analysis")
                        fp_finding.title = f"[FP] {fp_finding.title}"
                        self.logger.log_message(f"Marked finding as FP: {fp_finding.title}")
                
                elif action == "update":
                    # Update existing finding with new evidence
                    prev_id = finding_data.get("previous_finding_id")
                    if prev_id and prev_id in findings_map:
                        existing = findings_map[prev_id]
                        # Update fields if provided
                        if "severity" in finding_data:
                            existing.severity = finding_data["severity"]
                        if "description" in finding_data:
                            existing.description = finding_data["description"]
                        if "poc_request_raw" in finding_data:
                            existing.poc_request_raw = finding_data["poc_request_raw"]
                        if "poc_response_raw" in finding_data:
                            existing.poc_response_raw = finding_data["poc_response_raw"]
                        if "verified_by_test_id" in finding_data:
                            existing.verified_by_test_id = finding_data["verified_by_test_id"]
                            existing.source = "verified"
                        self.logger.log_message(f"Updated finding: {existing.title}")
                
                elif action == "create":
                    # Create new finding
                    finding = self._create_finding_from_result(finding_data)
                    
                    # Mark as verified if linked to test
                    if finding_data.get("verified_by_test_id"):
                        finding.source = "verified"
                        finding.verified_by_test_id = finding_data["verified_by_test_id"]
                    else:
                        finding.source = "inferred"
                    
                    analysis.findings.append(finding)
                    self.logger.log_message(f"Created new finding: {finding.title}")
        
        # Ensure we keep all previous findings (with updates applied)
        analysis.findings = list(findings_map.values()) + [
            f for f in analysis.findings if f.id not in findings_map
        ]
        
        # Update cached findings count
        analysis.findings_count = len(analysis.findings)
        self.logger.log_message(f"Incremental analysis complete: {analysis.findings_count} total findings")
        
        return analysis
    
    def _build_context_continuation_prompt(self, analysis: Analysis) -> str:
        """
        Build prompt for continuing analysis with user-provided context.
        
        Args:
            analysis: Analysis with user_context and existing findings
            
        Returns:
            Prompt string
        """
        prompt_template = self._load_prompt("bac_context_continuation")
        
        # Format existing findings summary
        findings_summary = ""
        if analysis.findings:

async def send_http_request(method: str, url: str, headers: dict = None, body: str = None, test_description: str = None) -> dict:

            import time
            start = time.time()
            
            response = await client.request(
                method=method.upper(),

                headers=headers or {{}},

            return {{
                "status_code": response.status_code,
                "headers": dict(response.headers),
                "body": response.text,

            }}

    def _parse_claude_response(self, response_text: str) -> Dict[str, Any]:
        """
        Parse Claude's response to extract structured data.
        Handles multiple JSON blocks by finding the one with expected fields.
        
        Args:
            response_text: Raw response text
            
        Returns:
            Parsed response dict
        """
        import re
        
        # Find all JSON code blocks
        json_blocks = []
        
        # Match ```json ... ``` blocks
        json_pattern = re.compile(r'```json\s*([\s\S]*?)```', re.MULTILINE)
        for match in json_pattern.finditer(response_text):
            try:
                parsed = json.loads(match.group(1).strip())
                json_blocks.append(parsed)
            except json.JSONDecodeError:
                continue
        
        # Also try ``` ... ``` blocks (without json tag)
        if not json_blocks:
            code_pattern = re.compile(r'```\s*([\s\S]*?)```', re.MULTILINE)
            for match in code_pattern.finditer(response_text):
                content = match.group(1).strip()
                if content.startswith('{'):
                    try:
                        parsed = json.loads(content)
                        json_blocks.append(parsed)
                    except json.JSONDecodeError:
                        continue
        
        # Find the block with expected fields
        expected_fields = ['test_cases', 'findings', 'analysis_id']
        for block in json_blocks:
            if isinstance(block, dict):
                if any(field in block for field in expected_fields):
                    return block
        
        # Fallback: return the largest JSON block (most likely to be the data)
        if json_blocks:
            return max(json_blocks, key=lambda x: len(str(x)))
        
        # Last resort: try to extract JSON from raw text
        try:
            if "{" in response_text:
                start = response_text.index("{")
                end = response_text.rindex("}") + 1
                return json.loads(response_text[start:end])
        except (json.JSONDecodeError, ValueError):
            pass
        
        return {"status": "unknown", "raw_response": response_text}

    def _validate_findings(self, findings: list, mcp_log_path: Path, test_results: dict) -> list:
        """
        Validate findings against actual MCP tool responses.
        
        Prevents hallucination by ensuring:
        1. MCP tool calls were actually made
        2. Responses don't contain rejection patterns
        3. Tool failures didn't occur
        
        Args:
            findings: List of Finding objects from Claude
            mcp_log_path: Path to MCP tool call log file
            test_results: Parsed test results from Claude
            
        Returns:
            List of validated Finding objects
        """
        # Check for tool_failure status in response
        if test_results.get("status") == "tool_failure":
            if self.logger:
                self.logger.log_error(
                    Exception(test_results.get("error", "MCP tool failure")),
                    "Analysis aborted due to MCP tool failure"
                )
            return []  # Reject all findings when tools failed
        
        # Check if MCP log exists
        if not mcp_log_path.exists():
            if self.logger:
                self.logger.log_message("WARNING: No MCP tool log found - findings may be hallucinated", "warning")
            # Allow findings but log the warning
            return findings
        
        # Load MCP tool log
        tool_calls = []
        try:
            with open(mcp_log_path) as f:
                for line in f:
                    if line.strip():
                        tool_calls.append(json.loads(line))
        except Exception as e:
            if self.logger:
                self.logger.log_error(e, "Failed to read MCP tool log")
            return findings  # Allow findings but log the error
        
        if len(tool_calls) == 0:
            if self.logger:
                self.logger.log_message("WARNING: MCP log is empty - no tool calls were made", "warning")
            # Could be hallucination, but don't block if there are findings
            return findings
        
        # Check for failed tool calls
        failed_calls = [tc for tc in tool_calls if not tc.get("success", True)]
        if failed_calls and self.logger:
            self.logger.log_message(f"WARNING: {len(failed_calls)} MCP tool calls failed", "warning")
        
        # Check for rejection patterns in tool calls
        secure_calls = [tc for tc in tool_calls if tc.get("result_summary", {}).get("is_rejection", False)]
        if self.logger:
            self.logger.log_message(
                f"MCP validation: {len(tool_calls)} calls, {len(secure_calls)} showed rejection patterns"
            )
        
        # Validate each finding - check if there are any non-rejection tool calls
        validated = []
        non_rejection_calls = [tc for tc in tool_calls 
                               if tc.get("success") and 
                               not tc.get("result_summary", {}).get("is_rejection", False)]
        
        if len(non_rejection_calls) == 0 and len(findings) > 0:
            # All tool calls showed rejection but Claude still reported findings
            if self.logger:
                self.logger.log_message(
                    f"REJECTING {len(findings)} findings: all {len(tool_calls)} tool calls showed rejection patterns",
                    "warning"
                )
            return []  # Reject all findings
        
        # Keep findings but log validation results
        for finding in findings:
            if self.logger:
                self.logger.log_message(f"Validated finding: {finding.title}", "debug")
            validated.append(finding)
        
        return validated
    
    def _extract_findings_from_text(self, text: str) -> list:
        """
        Extract findings from markdown/text format as fallback.
        Used when Claude returns findings in text format instead of JSON.
        
        Args:
            text: Claude's text response
            
        Returns:
            List of Finding objects
        """
        import re
        findings = []
        
        if self.logger:
            self.logger.log_message("Attempting to extract findings from text format")
        
        # Pattern to match findings like "### **1. Missing Authorization Check (HIGH)**"
        # or "**1. Missing Authorization Check**" or "1. Missing Authorization (HIGH)"
        pattern = r'\*\*\d+\.\s+(.+?)\s*\(?(HIGH|CRITICAL|MEDIUM|LOW)?\)?\*\*'
        matches = re.findall(pattern, text, re.IGNORECASE)
        
        for title, severity in matches:
            title = title.strip()
            severity_value = severity.lower() if severity else "medium"
            
            # Determine vulnerability type from title
            vuln_type = "unknown"
            title_lower = title.lower()
            if "idor" in title_lower or "direct object" in title_lower:
                vuln_type = "idor"
            elif "auth" in title_lower or "authorization" in title_lower:
                vuln_type = "auth_bypass"
            elif "privilege" in title_lower or "escalation" in title_lower:
                vuln_type = "privilege_escalation"
            elif "path" in title_lower or "traversal" in title_lower:
                vuln_type = "path_traversal"
            
            finding = Finding(
                vulnerability_type=vuln_type,
                severity=severity_value,
                title=title,
                description=f"Extracted from AI analysis. See AI analysis field for full details.",
                injection_point="See AI analysis for details",
                remediation="See AI analysis for remediation recommendations"
            )
            findings.append(finding)
            
            if self.logger:
                self.logger.log_message(f"Extracted finding: {title} ({severity_value})")
        
        return findings

        # Check category-specific prompt first (e.g., prompts/bac/bac_initial_analysis.txt)
        category_prompt = self.prompts_dir / "bac" / f"{prompt_name}.txt"
        if category_prompt.exists():
            return category_prompt.read_text()
        
        # Check shared prompts (e.g., prompts/shared/vulnerability_detection.txt)
        shared_prompt = self.prompts_dir / "shared" / f"{prompt_name}.txt"
        if shared_prompt.exists():
            return shared_prompt.read_text()
        
        # Return default prompt if file not found
        return "Analyze this request for BAC/IDOR vulnerabilities: {request}"
    
    def _save_analysis_state(self, analysis: Analysis, session_dir: Path):
        """
        Save complete analysis state for debugging.
        
        Args:
            analysis: Analysis object
            session_dir: Session directory
        """
        state_file = session_dir / "analysis_state.json"
        
        try:
            # Helper to convert datetime objects to ISO format strings
            def serialize_model(model):
                """Serialize a Pydantic model with datetime handling."""
                if model is None:
                    return None
                data = model.dict()
                # Convert any datetime fields to ISO strings
                for key, value in data.items():
                    if hasattr(value, 'isoformat'):
                        data[key] = value.isoformat()
                return data
            
            # Convert to dict
            state = {
                "id": analysis.id,
                "status": analysis.status.value,
                "target_url": analysis.target_url,
                "created_at": analysis.created_at.isoformat(),
                "updated_at": analysis.updated_at.isoformat(),
                "error": analysis.error,
                "ai_analysis": analysis.ai_analysis,
                "auth_profile": serialize_model(analysis.auth_profile),
                "findings_count": len(analysis.findings),
                "bac_tests_count": len(analysis.bac_tests),
                "data_requests": [serialize_model(dr) for dr in analysis.data_requests],
                "related_requests_count": len(analysis.related_requests),
            }
            
            with open(state_file, 'w') as f:
                json.dump(state, f, indent=2)
            
            if self.logger:
                self.logger.log_message(f"Analysis state saved to {state_file}", "debug")
        
        except Exception as e:
            if self.logger:
                self.logger.log_error(e, "Failed to save analysis state")
    
    async def analyze_with_deterministic_testing(self, analysis: Analysis) -> Analysis:
        """
        Three-phase deterministic analysis.
        
        Phase 1: Claude generates test cases (no HTTP execution)
        Phase 2: Python executes tests deterministically via httpx
        Phase 3: Claude analyzes actual test results
        
        This approach prevents hallucination by ensuring:
        - All HTTP requests are actually executed
        - Responses are real, not inferred
        - Findings are marked as "verified" or "inferred"
        
        Args:
            analysis: Analysis object with initial request/response pairs
            
        Returns:
            Updated analysis with verified findings
        """
        # Create logger for this analysis
        self.logger = AnalysisLogger(analysis.id, self.work_dir)
        self.logger.log_message("Starting deterministic three-phase analysis")
        
        session_dir = self.work_dir / analysis.id
        session_dir.mkdir(exist_ok=True)
        
        try:
            # === PHASE 1: Generate Test Plan ===
            self.logger.log_message("=== PHASE 1: Generating test plan ===")
            test_plan = await self._generate_test_plan(analysis, session_dir)
            
            if not test_plan or not test_plan.get("test_cases"):
                self.logger.log_message("No test cases generated, completing with inferred findings only")
                analysis.status = AnalysisStatus.COMPLETED
                analysis.completed_at = datetime.utcnow()
                self._save_analysis_state(analysis, session_dir)
                return analysis
            
            self.logger.log_message(f"Generated {len(test_plan.get('test_cases', []))} test cases")
            
            # Save test plan
            test_plan_path = session_dir / "test_cases.json"
            with open(test_plan_path, 'w') as f:
                json.dump(test_plan, f, indent=2)
            
            # === PHASE 2: Execute Tests Deterministically ===
            self.logger.log_message("=== PHASE 2: Executing tests (no AI involvement) ===")
            
            from engine.core.test_executor import TestExecutor
            executor = TestExecutor(analysis.id, self.work_dir)
            test_results = executor.execute_all_tests()
            
            if not test_results or not test_results.get("test_results"):
                self.logger.log_message("No test results generated")
                analysis.status = AnalysisStatus.COMPLETED
                analysis.completed_at = datetime.utcnow()
                self._save_analysis_state(analysis, session_dir)
                return analysis
            
            self.logger.log_message(f"Executed {len(test_results.get('test_results', []))} tests")
            
            # === PHASE 3: Analyze Results ===
            self.logger.log_message("=== PHASE 3: Analyzing test results ===")
            analysis = await self._analyze_test_results(analysis, test_results, session_dir)
            
            # Attach raw request/response to verified findings
            for finding in analysis.findings:
                if finding.source == "verified" and finding.verified_by_test_id:
                    test_result = next(
                        (t for t in test_results.get("test_results", [])
                         if t.get("test_id") == finding.verified_by_test_id),
                        None
                    )
                    if test_result:
                        finding.poc_request_raw = test_result.get("request_raw")
                        finding.poc_response_raw = test_result.get("response_raw")
                        self.logger.log_message(
                            f"Attached Burp-friendly PoC to finding: {finding.title[:50]}",
                            "debug"
                        )
            
            analysis.status = AnalysisStatus.COMPLETED
            analysis.completed_at = datetime.utcnow()
            self._save_analysis_state(analysis, session_dir)
            
            self.logger.log_message(
                f"Deterministic analysis complete: {len(analysis.findings)} findings "
                f"({sum(1 for f in analysis.findings if f.source == 'verified')} verified, "
                f"{sum(1 for f in analysis.findings if f.source == 'inferred')} inferred)"
            )
            
            return analysis
        
        except Exception as e:
            self.logger.log_error(e, "Deterministic analysis failed")
            analysis.status = AnalysisStatus.FAILED
            analysis.error = f"Deterministic analysis failed: {str(e)}"
            self._save_analysis_state(analysis, session_dir)
            return analysis
    
    async def _generate_test_plan(self, analysis: Analysis, session_dir: Path) -> Dict[str, Any]:
        """
        Phase 1: Generate test cases without executing HTTP requests.
        
        Args:
            analysis: Analysis object with captured traffic
            session_dir: Session directory
            
        Returns:
            Test plan dict with test_cases array
        """
        if not ClaudeSDKClient:
            raise Exception("ClaudeSDKClient not available")
        
        # Add Claude CLI to PATH if needed
        claude_cli_dir = os.path.expanduser(
            os.getenv("CLAUDE_CLI_DIR", "~/.claude/local")
        )
        if claude_cli_dir and os.path.exists(claude_cli_dir):
            os.environ["PATH"] = f"{claude_cli_dir}:{os.environ.get('PATH', '')}"
        
        # Configure Claude options - NO MCP tools for test generation
        # Removed Write/Edit to prevent Claude from writing to files instead of returning inline
        options = ClaudeCodeOptions(
            cwd=str(session_dir),
            allowed_tools=["Bash", "Glob", "Grep", "LS", "Read"],
            permission_mode="acceptEdits"
        )
        
        # Build prompt
        prompt_template = self._load_prompt("bac_generate_tests")
        
        # Format captured traffic
        if analysis.request_response_pairs:
            request_info, response_info = self._format_request_response_pairs(analysis)
        else:
            request_info = self._format_single_request(analysis.request)
            response_info = self._format_single_response(analysis.response) if analysis.response else ""
        
        prompt = prompt_template.replace("{captured_traffic}", request_info + "\n" + response_info)
        prompt = prompt.replace("{subdomain}", analysis.subdomain or "unknown")
        prompt = prompt.replace("{analysis_id}", analysis.id)
        prompt = prompt.replace("{user_context}", analysis.user_context or "(No additional context provided)")
        
        self.logger.log_query(prompt)
        
        async with ClaudeSDKClient(options=options) as client:
            await client.query(prompt)
            response_text = await self._collect_response(client)
            self.logger.log_response(response_text)
        
        # Parse test plan
        test_plan = self._parse_claude_response(response_text)
        
        # Fallback: check if Claude wrote to a file instead
        if not test_plan.get("test_cases"):
            test_plan_file = session_dir / "test_plan.json"
            if test_plan_file.exists():
                self.logger.log_message("Found test_plan.json file, loading from disk", "debug")
                try:
                    with open(test_plan_file) as f:
                        test_plan = json.load(f)
                except (json.JSONDecodeError, IOError) as e:
                    self.logger.log_message(f"Failed to load test_plan.json: {e}", "warning")
        
        return test_plan
    
    async def _analyze_test_results(
        self,
        analysis: Analysis,
        test_results: Dict[str, Any],
        session_dir: Path
    ) -> Analysis:
        """
        Phase 3: Analyze actual test results and generate verified findings.
        
        Args:
            analysis: Analysis object
            test_results: Results from TestExecutor
            session_dir: Session directory
            
        Returns:
            Updated analysis with verified findings
        """
        if not ClaudeSDKClient:
            raise Exception("ClaudeSDKClient not available")
        
        # Add Claude CLI to PATH if needed
        claude_cli_dir = os.path.expanduser(
            os.getenv("CLAUDE_CLI_DIR", "~/.claude/local")
        )
        if claude_cli_dir and os.path.exists(claude_cli_dir):
            os.environ["PATH"] = f"{claude_cli_dir}:{os.environ.get('PATH', '')}"
        
        # Configure Claude options - NO MCP tools for result analysis
        options = ClaudeCodeOptions(
            cwd=str(session_dir),
            allowed_tools=["Bash", "Glob", "Grep", "LS", "Read", "Write", "Edit"],
            permission_mode="acceptEdits"
        )
        
        # Build prompt
        prompt_template = self._load_prompt("bac_analyze_results")
        
        # Format test results
        test_results_json = json.dumps(test_results, indent=2)
        
        prompt = prompt_template.replace("{test_results}", test_results_json)
        prompt = prompt.replace("{analysis_id}", analysis.id)
        
        self.logger.log_query(prompt)
        
        async with ClaudeSDKClient(options=options) as client:
            await client.query(prompt)
            response_text = await self._collect_response(client)
            self.logger.log_response(response_text)
        
        # Parse results
        result = self._parse_claude_response(response_text)
        analysis.ai_analysis = response_text
        
        # Claude writes findings to verified_findings.json via file operations
        # We don't parse from response anymore - just update the cached count
        verified_findings_file = session_dir / "verified_findings.json"
        if verified_findings_file.exists():
            try:
                with open(verified_findings_file, 'r') as f:
                    findings_data = json.load(f)
                analysis.findings_count = len(findings_data.get("findings", []))
                self.logger.log_message(f"Found {analysis.findings_count} findings in verified_findings.json")
            except Exception as e:
                self.logger.log_message(f"Warning: Could not count findings: {e}", "warning")
                analysis.findings_count = 0
        else:
            analysis.findings_count = 0
            self.logger.log_message("No verified_findings.json file found")
        
        return analysis
